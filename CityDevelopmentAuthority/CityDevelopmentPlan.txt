public class CityBuilder
{
    public void BuildCity()
    {
        CreateBuildingZones();
        SetUpRoadsAndTraffic();
        InstallPublicTransportSystem();
        SetUpHealthcareFacilities();
        SetUpPoliceAndLawServices();
    }

    private void CreateBuildingZones()
    {
        SetResidentialZones();
        SetCommercialZones();
        SetIndustrialZones();
        BuildBuildings();
    }

    private void SetUpRoadsAndTraffic()
    {
        DesignRoadNetwork();
        BuildRoadsAndStreets();
        ManageTrafficSignals();
        DesignPedestrianPaths();
        SetUpSmartTrafficManagement();
    }

    private void InstallPublicTransportSystem()
    {
        SetUpBusNetwork();
        SetUpMetroNetwork();
        BuildPublicTransportStations();
    }

    private void SetUpHealthcareFacilities()
    {
        BuildHospitals();
        SetUpHealthcareClinics();
        SetUpMobileHealthcareUnits();
    }

    private void SetUpPoliceAndLawServices()
    {
        SetUpPoliceStations();
        SetUpEmergencyResponseUnits();
		SetUpSessionCourts();
    }
}
